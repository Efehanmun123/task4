    cin >> girilen;
    vector sayi;
    vector islem;
    string sayitext = "";
    string islemler = "^*/+-%b";
    for (int i = 0; i < girilen.length(); i++) {
        bool bulundu = false;
        for (int j = 0; j < islemler.length(); j++) {
            if (girilen[i] == islemler[j]) {
                string c(1, girilen[i]);

                islem.push_back(c);
                sayi.push_back(strtod(sayitext.c_str()));
                sayitext = "";
                bulundu = true;
            }
        }
        if (bulundu == false) {
            sayitext += girilen[i];
        }
    }
    sayi.push_back(strtod(sayitext.c_str()));
    double  y = 0;
    auto calculate = [&](double n, string op) {
        if (op == "*") {
            y = sayi[n] * sayi[n + 1];
        }
        if (op == "/") {
            y = sayi[n] / sayi[n + 1];
        }
        if (op == "+") {
            y = sayi[n] + sayi[n + 1];
        }
        if (op == "-") {
            y = sayi[n] - sayi[n + 1];
        }
        if (op == "%") {
            y = fmod(sayi[n], sayi[n + 1]);
        }
        if (op == "^") {
            y = pow(sayi[n], sayi[n + 1]);
        }
        if (op == "b") {
            int ondalik = 1;
            int convert = 0;
            while (sayi[n] != 0) {
                int rem = fmod(sayi[n], sayi[n + 1]);
                convert += rem * ondalik;
                sayi[n] /= sayi[n + 1];
                ondalik *= 10;
            }
            y = convert;
        }
        islem.erase(islem.begin() + n);
        sayi[n] = y;
        sayi.erase(sayi.begin() + (n + 1));
    };
    int i = 0;
    while (i < islem.size()) {

        if (islem[i] == "^") {
            calculate(i, islem[i]);
            i -= 1;
        }
        i++;
    }
    i = 0;
    while (i < islem.size()) {
        if (islem[i] == "*" || islem[i] == "/") {
            calculate(i, islem[i]);
            i -= 1;
        }
        i++;
    }
    i = 0;
    while (i < islem.size()) {
        if (islem[i] == "+" || islem[i] == "-") {
            calculate(i, islem[i]);
            i -= 1;
        }
        i++;
    }
    i = 0;
    while (i < islem.size()) {
        if (islem[i] == "%") {
            calculate(i, islem[i]);
        }
        i++;
    }
    if (islem.size() > 0) {
        if (islem[0] == "b") {
            calculate(0, islem[0]);
        }

    }
